    Class Polyline

    A continuous set of lines.

    Examples
    var a = new Vector3();
var b = new Vector3(10, 10);
var c = new Vector3(20, 5);
var d = new Vector3(25, 10);

var pline = new Polyline(new[] { a, b, c, d });
var offset = pline.Offset(1, EndType.Square);

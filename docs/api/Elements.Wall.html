    Class Wall

    A wall defined by a planar profile extruded to a height.

    Examples
    // Create some openings.
var openings = new List<Opening>(){
    new Opening(1.0, 2.0, 1.0, 1.0),
    new Opening(3.0, 1.0, 1.0, 2.0)
};
wall.Openings.AddRange(openings);
